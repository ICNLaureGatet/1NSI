def extractionDonnees(nomFichier):
    """Cette fonction rcupre les donnes d'un fichier csv et renvoie un dictionnaire :
    La liste des descripteurs et la liste de toutes les donnes"""
    nomDuFichier = nomFichier
    if nomFichier[-4:]!=".csv":
        nomDuFichier+=".csv"
    fichier = open(nomDuFichier,"r",encoding="utf-8")
    extraction=csv.DictReader(fichier,delimiter=";")
    listeJoueurs=[]
    for ligne in extraction:
        listeJoueurs.append(dict(ligne))
    return listeJoueurs

1. Le principe de lalgorithme en langage courant

On dispose dun ensemble de donnes, dune fonction distance et dun entier k. Chaque donne contient deux types dinformations :

  • deux donnes numriques destines la comparaison de deux lments de lensemble;
  • un troisime critre destin la classification dun lment.

from math import sqrt
def distance(x1,y1,x2,y2):
    """distance euclidienne entre les points de coordonnes (x1;y1) et (x2;y2)"""
    return sqrt((x2-x1)**2+(y2-y1)**2)

(c) Le principe de lalgorithme

  1. Parcourir la liste de tous les joueurs de l'quipe choisie (c'est une liste de dictionnaires dans notre exemple) ;
  2. Calculer pour chacun des joueurs la "distance" (en termes de taille et poids) avec le nouveau joueur ;
  3. Rcuprer les k joueurs les plus proches en termes de distance ;
  4. Dans cette liste de k joueurs, identifier le critre de "type de poste" le plus souvent prsent et l'affecter au nouveau joueur.

(d) Avec le tableur
Le problme avec la formule propose c'est qu'en la recopiant vers le bas, les rfrences I2 et J2 voluent et les formules recopies crent alors des erreurs.

def classification(k,data,taille,poids):
    """une srie de donnes tant fournies, on souhaite extraire la classe type de poste d'un joueur dont la taille et le poids sont fournis
    data est une liste de dictionnaires de cls : 'Equipe', 'Nom', 'Poste', 'Type poste', 'Date de naissance', 'Taille', 'Poids'
    On utilise l'algorithme des k plus proches voisins avec k premier paramtre de cette fonction
    Cet algorithme utilise un tri des donnes"""
    knn = sorted(data, key=lambda x:distance(int(x["Taille (en cm)"]),int(x["Poids (en kg)"]),taille,poids))[:k] #on ordonne les lments de la liste data suivant la distance (key) et on en prend que les k premiers lments
    return rechercheTypePoste(knn)

def classificationEfficace(k,data,taille,poids):
    """une srie de donnes tant fournies, on souhaite extraire la classe type de poste d'un joueur dont la taille et le poids sont fournis
    data est une liste de dictionnaires de cls : 'Equipe', 'Nom', 'Poste', 'Type poste', 'Date de naissance', 'Taille', 'Poids'
    On utilise l'algorithme des k plus proches voisins avec k premier paramtre de cette fonction
    Cet algorithme ne range pas les donnes dans l'ordre"""
    extraction = [] # cette liste contiendra un tuple o le premier lment sera la ligne d'un joueur et le deuxime lment la distance par rapport au nouveau joueur (cela vite de la recalculer)
    for joueur in data:
        newDist = distance(int(joueur['Taille (en cm)']),int(joueur["Poids (en kg)"]),taille,poids)
        aPlacer = len(extraction)<k or newDist<extraction[-1][1]    #au premier passage, extraction est vide, mais extraction[-1] n'est pas valu donc a ne plante pas
        #ce boolen aPlacer me permet de dterminer si j'ajoute le joueur  ma liste des k plus proches voisins
        if aPlacer:
            #On met joueur dans la liste extraction
            #On le met cependant  sa place dans la liste ordonne par les distances
            if(len(extraction)==0):
                extraction.append((joueur, newDist)) #J'ajoute un tuple, donc j'ai besoin de ces deux parenthses
            else:
                if(len(extraction)==k):
                    extraction.pop() #avant d'ajouter un nouvel lment, on enlve le dernier
                pos=len(extraction)
                while (pos>=1 and newDist<extraction[pos-1][1]):
                    pos-=1
                extraction.insert(pos,(joueur,newDist)) #l'lment est alors insr  sa place (dans l'ordre croissant de la distance)
    knn = [el[0] for el in extraction]
    return rechercheTypePoste(knn)
